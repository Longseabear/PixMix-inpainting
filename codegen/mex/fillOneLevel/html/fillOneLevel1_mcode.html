1function [filledI, retf] = fillOneLevel(initf, I, M, level, useLineConstr)
  2% initf is initial value of the mapping f
  3% Use random sampling and propagation to fill the missing parts of I with
  4% the known pixels.
  5
  6% use 8-neighbor for cost_spatial, 
  7% use 5x5 patch for cost_apperance.
  8
  9global f;
 10global R;
 11global C;
 12global im;
 13global mask;
 14f = initf;
 15numIter = 5 ;
 16[R, C] = size(M);
 17im = I;
 18mask = M;
 19D = bwdist(~M);
 20alphaSp = 0.05;
 21alphaAp = 0.95;
 22wrs = 0.5; 
 23rrs = min(R,C); % weight random search and range random search
 24%% fill the image with initial guess of f
 25for i=1:R
 26    for j=1:C
 27        if M(i,j) == 1
 28            im(i,j,:) = im(f(i,j,1), f(i,j,2), :);
 29        end
 30    end
 31end
 32
 33%% the propagation and random search
 34for iter = 1:numIter
 35    for i=1:R
 36        for j=1:C
 37            if M(i,j) == 1
 38                im(i,j,:) = im(f(i,j,1), f(i,j,2), :);
 39            end
 40        end
 41    end
 42%     imshow(im);
 43    if mod(iter,2) == 1
 44        for i=1:R
 45            for j=1:C
 46                %% propagation
 47                if M(i,j)==1
 48                    fijOld = squeeze(f(i,j,:));
 49                    costLeft = Inf;
 50                    costUp = Inf;
 51                    spcOld = calcSpatialCost(i,j, fijOld);
 52                    apcOld = calcAppearanceCost(i,j, fijOld);
 53                    costOld = spcOld*alphaSp + apcOld*alphaAp;
 54                    if M(i-1, j)==1 && i-1>0 && isValid(squeeze(f(i-1,j,:))+[1;0])
 55                        fijLeft = squeeze(f(i-1,j,:))+[1;0];
 56                        spcLeft = calcSpatialCost(i,j, fijLeft);
 57                        apcLeft = calcAppearanceCost(i,j, fijLeft);
 58                        costLeft = spcLeft*alphaSp + apcLeft*alphaAp;
 59                    end
 60                    if M(i, j-1)==1 && j-1>0 && isValid(squeeze(f(i,j-1,:))+[0;1])
 61                        fijUp = squeeze(f(i,j-1,:))+[0;1];
 62                        spcUp = calcSpatialCost(i,j, fijUp);
 63                        apcUp = calcAppearanceCost(i,j, fijUp);
 64                        costUp = spcUp*alphaSp + apcUp*alphaAp;
 65                    end
 66                    
 67                    if costLeft < min(costUp, costOld)
 68                        f(i,j,:) = fijLeft;
 69                        costOld = costLeft;
 70                    elseif costUp < min(costLeft, costOld)
 71                        f(i,j,:) = fijUp;
 72                        costOld = costUp;
 73                    end
 74                    
 75                    %% random search
 76                    drs = wrs*rrs;
 77                    while(drs>D(i,j)+3)
 78                        searchvec = round((rand(1,2)-0.5)*drs) + [i,j];
 79                        num = 3;
 80                        while(~isValid(searchvec) && num>0)
 81                            searchvec = round((rand(1,2)-0.5)*drs) + [i,j];
 82                            num = num-1;
 83                        end
 84                        fijRS = searchvec';
 85                        spcRS = calcSpatialCost(i,j, fijRS);
 86                        apcRS = calcAppearanceCost(i,j, fijRS);
 87                        costRS = spcRS*alphaSp + apcRS*alphaAp;
 88                        
 89                        if costRS<costOld
 90                            f(i,j,:) = fijRS;
 91                            costOld = costRS;
 92                        end
 93                        
 94                        drs = drs*wrs;
 95                    end
 96                end
 97                
 98            end % end for j
 99        end % end for i
100        
101    else % iter is even
102        for i=R:-1:1
103            for j=C:-1:1
104                
105                if M(i,j)==1
106                    fijOld = squeeze(f(i,j,:));
107                    costRight = Inf;
108                    costDown = Inf;
109                    spcOld = calcSpatialCost(i,j, fijOld);
110                    apcOld = calcAppearanceCost(i,j, fijOld);
111                    costOld = spcOld*alphaSp + apcOld*alphaAp;
112                    if M(i+1, j)==1 && i+1<=R && isValid(squeeze(f(i+1,j,:))+[-1;0])
113                        fijRight = squeeze(f(i+1,j,:))+[-1;0];
114                        spcRight = calcSpatialCost(i,j, fijRight);
115                        apcRight = calcAppearanceCost(i,j, fijRight);
116                        costRight = spcRight*alphaSp + apcRight*alphaAp;
117                    end
118                    if M(i, j+1)==1 && j+1>0 && isValid(squeeze(f(i,j+1,:))+[0;-1])
119                        fijDown = squeeze(f(i,j+1,:))+[0;-1];
120                        spcDown = calcSpatialCost(i,j, fijDown);
121                        apcDown = calcAppearanceCost(i,j, fijDown);
122                        costDown = spcDown*alphaSp + apcDown*alphaAp;
123                    end
124                    
125                    if costRight < min(costDown, costOld)
126                        f(i,j,:) = fijRight;
127                    elseif costDown < min(costRight, costOld)
128                        f(i,j,:) = fijDown;
129                    end
130                    %% random search
131                    drs = wrs*rrs;
132                    while(drs>D(i,j))
133                        searchvec = round((rand(1,2)-0.5)*drs) + [i,j];
134                        num = 3;
135                        while(~isValid(searchvec) && num>0)
136                            searchvec = round((rand(1,2)-0.5)*drs) + [i,j];
137                            num = num-1;
138                        end
139                        fijRS = searchvec';
140                        spcRS = calcSpatialCost(i,j, fijRS);
141                        apcRS = calcAppearanceCost(i,j, fijRS);
142                        costRS = spcRS*alphaSp + apcRS*alphaAp;
143                        
144                        if costRS<costOld
145                            f(i,j,:) = fijRS;
146                            costOld = costRS;
147                        end
148                        
149                        drs = drs*wrs;
150                    end
151                end
152                
153            end % end for j
154        end % end for i
155        
156    end % end if
157    
158end
159retf = f;
160filledI = im;
161
162end
163
164% fij may change due to propagation or random search.
